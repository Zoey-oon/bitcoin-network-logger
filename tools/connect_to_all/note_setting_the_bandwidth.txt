VBoxManage bandwidthctl "Bitcoin Minimum" add Limit --type network --limit 20k
VBoxManage modifyvm "Bitcoin Minimum" --nicbandwidthgroup1 Limit
VBoxManage modifyvm "Bitcoin Minimum" --nicbandwidthgroup2 Limit


# Bitcoin minimum: 250k per day = 250*8/24/60/60


To set a new amount:
VBoxManage bandwidthctl "Bitcoin Minimum" set Limit --limit 100k
